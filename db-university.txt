Query con Grup by
1. Contare quanti iscritti ci sono stati ogni anno
SELECT YEAR(`enrolment_date`) AS 'anno', COUNT(`id`) AS 'numero_iscritti'
FROM students
GROUP BY YEAR(enrolment_date);

2. Contare gli insegnanti che hanno l'ufficio nello stesso edificio
SELECT `office_address` AS 'ufficio', COUNT(`id`) AS 'numero_professori'
FROM `teachers`
GROUP BY `office_address`;

3. Calcolare la media dei voti di ogni appello d'esame
SELECT `exam_id` AS 'numero_esame', AVG(`vote`) AS 'voto_medio'
FROM `exam_student`
GROUP BY `exam_id`;

4. Contare quanti corsi di laurea ci sono per ogni dipartimento
SELECT COUNT(`department_id`) AS 'numero_di_corsi_di_laurea', `department_id` AS 'numero_dipartimento'
FROM `degrees`
GROUP BY `department_id`;
